**2. Create decorator and query interface:**

`Python/structural_lib/clause_db.py`:
```python
"""IS 456 Code Clause Database and Traceability System."""

import json
from pathlib import Path
from typing import Any
from functools import wraps

# Load clause database
_CLAUSE_DB_PATH = Path(__file__).parent / "data" / "is456_clauses.json"
_CLAUSE_DB: dict[str, Any] = {}

def load_clause_db():
    """Load IS 456 clause database from JSON."""
    global _CLAUSE_DB
    if not _CLAUSE_DB:
        with open(_CLAUSE_DB_PATH) as f:
            data = json.load(f)
            _CLAUSE_DB = {c["clause_id"]: c for c in data["clauses"]}
    return _CLAUSE_DB

def query(search_term: str) -> list[dict[str, Any]]:
    """
    Search for clauses by keyword in tags, title, or requirement.

    Args:
        search_term: Keyword to search (e.g., "minimum steel", "deflection")

    Returns:
        List of matching clause dictionaries
    """
    db = load_clause_db()
    results = []
    search_lower = search_term.lower()

    for clause_id, clause in db.items():
        if (
            search_lower in clause["title"].lower()
            or search_lower in clause["requirement"].lower()
            or any(search_lower in tag for tag in clause["tags"])
        ):
            results.append(clause)

    return results

def get_clause(clause_id: str) -> dict[str, Any] | None:
    """Get specific clause by ID (e.g., '26.5.1.1')."""
    db = load_clause_db()
    return db.get(clause_id)

def clause(*clause_ids: str):
    """
    Decorator to mark function with IS 456 clause references.

    Usage:
        @clause("26.5.1.1", "26.5.2")
        def calculate_min_steel(b: float, d: float, fy: float) -> float:
            ...
    """
    def decorator(func):
        func.__clause_refs__ = clause_ids

        @wraps(func)
        def wrapper(*args, **kwargs):
            return func(*args, **kwargs)

        return wrapper
    return decorator

def get_function_clauses(func) -> tuple[str, ...]:
    """Get clause references from decorated function."""
    return getattr(func, "__clause_refs__", ())
```
